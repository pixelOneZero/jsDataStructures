Array.prototype.filter()

Purpose: Returns an array of unique elements from the array provided as input.

Input:

var advertisers = ["time warner","target","target","kohl's","target","nordstrom","cvs","nordstrom","sephora","walgreen's"];

Example Usage:

var uniqueAdvertisers = advertisers.filter(function(elem, i, array){ return advertisers.indexOf(elem) === i; });

Array.prototype.map()

Purpose: Returns an array of transformed elements corresponding to elements in the input array.

Input:

var kilometers = [10, 30, 92, 53, 31, 79, 41, 49, 51, 91];

Example Usage:

var miles = kilometers.map(function(elem, i) { return elem * 0.621371; });

Array.prototype.reduce()

Purpose: Returns an aggregate value based on elements in a provided array or object.

Input:

var topSellers = [{ 'title': 'The Mythical Man Month', 'units': 3791 }, { 'title': 'The Mythical Man Month', 'units': 2703 }, { 'title': 'Eloquent JavaScript', 'units': 2889 }, { 'title': 'JavaScript: The Good Parts', 'units': 3016 }, { 'title': 'Secrets of the JavaScript Ninja', 'units': 4082 }];

Example Usage:

var totalUnits = topSellers.reduce(function(previousValue, element) { return previousValue + element.units; }, 0);
